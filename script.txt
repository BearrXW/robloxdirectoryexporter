-- Roblox Studio Script
-- Lists every object inside visible Explorer services (including Workspace)
-- Skips hidden/engine-only services
-- Copies all full paths + class names to clipboard

local visibleServices = {
	"Workspace",
	"ReplicatedStorage",
	"ServerScriptService",
	"ServerStorage",
	"StarterGui",
	"StarterPack",
	"StarterPlayer",
	"Lighting",
	"ReplicatedFirst",
	"SoundService",
	"Players",
}

local paths = {}

-- Function to get a clean full path
local function getFullPath(obj)
	local path = obj.Name
	local parent = obj.Parent
	while parent and parent ~= game do
		path = parent.Name .. "/" .. path
		parent = parent.Parent
	end
	return parent and "game/" .. path or path
end

-- Collect all instances inside the visible services
for _, serviceName in ipairs(visibleServices) do
	local service = game:FindFirstChild(serviceName)
	if service then
		table.insert(paths, getFullPath(service) .. " (" .. service.ClassName .. ")")
		for _, obj in ipairs(service:GetDescendants()) do
			table.insert(paths, getFullPath(obj) .. " (" .. obj.ClassName .. ")")
		end
	end
end

-- Sort alphabetically
table.sort(paths)

local output = table.concat(paths, "\n")

print("Found " .. #paths .. " visible objects. Copying to clipboard...")

if setclipboard then
	setclipboard(output)
	print("✅ Copied all visible service paths (including Workspace).")
else
	print("⚠️ setclipboard() not available in this environment.")
end
